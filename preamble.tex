%.......................... current value for width of the gifure

% particularly small figures but it should be reset immediately after
% the

\newlength{\figwidth}
\setlength{\figwidth}{\figmaxwidth}

\centerline{\psfig{figure=#2.eps,width=\figwidth}}
\\ \hline \end{tabular}
\refstepcounter{figure}\label{#4}

\end{quote}

}
